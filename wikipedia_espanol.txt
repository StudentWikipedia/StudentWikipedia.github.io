      <script>
          function runSearch() {
              // 1. Get the word the user typed
              var animal = document.getElementById("search-box").value.toLowerCase();

              // 2. Go to that file! (Example: if they type "Panda", go to panda.html)
              if (animal != "") {
                  window.location.href = "./data-release-espanol/" + animal + ".html";
              }
          }

          // Make the "Search" button run the code when clicked
          document.getElementById("search-button").onclick = runSearch;
          document.getElementById("search-box").onkeypress = function (e) {
              if (e.keyCode == 13) { runSearch(); }
          };
      </script>
